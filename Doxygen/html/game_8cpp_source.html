132  // resulting multiplicity of direction. If a component is set to -1, it

135 
136  // ball is beyond left side of table's bounds
137  if (bPos.x() - ball->getRadius() <= 0) {

142  // flip velocity if wrong dir
143  if (ball->getVelocity().x() >= 0) vChange.setX(-1);
144  }
145  // ball is above top of the table's bounds
146  if (bPos.y() - ball->getRadius() <= 0) {
147  // flip iff we're travelling in the wrong dir
148  if (ball->getVelocity().y() <= 0) vChange.setY(-1);
149  // ball is beyond bottom of table's bounds

152  if (ball->getVelocity().y() >= 0) vChange.setY(-1);

161 std::pair<QVector2D, QVector2D> Game::resolveCollision(Ball* ballA, Ball* ballB) {
162  // SOURCE : ASSIGNMENT SPEC
163 
164  // if not colliding (distance is larger than radii)
165  QVector2D collisionVector = ballB->getPosition() - ballA->getPosition();
166  if (collisionVector.length() > ballA->getRadius() + ballB->getRadius()) {
167  throw std::logic_error("attempting to resolve collision of balls that do not touch");
168  }
169  collisionVector.normalize();
170 
171  QVector2D ballAStartingVelocity = ballA->getVelocity();
172  QVector2D ballBStartingVelocity = ballB->getVelocity();
173 
174  float mr = ballB->getMass() / ballA->getMass();
175  double pa = QVector2D::dotProduct(collisionVector, ballA->getVelocity());
176  double pb = QVector2D::dotProduct(collisionVector, ballB->getVelocity());
177 
178  if (pa <= 0 && pb >= 0) return std::make_pair(QVector2D(0,0), QVector2D(0,0));
179 
180  double a = -(mr + 1);
181  double b = 2*(mr * pb + pa);
182  double c = -((mr - 1)*pb*pb + 2*pa*pb);
183  double disc = sqrt(b*b - 4*a*c);
184  double root = (-b + disc)/(2*a);
185  if (root - pb < 0.01) {
186  root = (-b - disc)/(2*a);
187  }
188 
189  ballA->changeVelocity(mr * (pb - root) * collisionVector);
190  ballB->changeVelocity((root-pb) * collisionVector);
191 
192  // return the change in velocities for the two balls
193  return std::make_pair(ballA->getVelocity() - ballAStartingVelocity, ballB->getVelocity() - ballBStartingVelocity);
194 }
